44  // initial voltage behind the transient reactance in the dq0 reference frame

63 void DP::Ph1::SynchronGenerator4OrderPCM::calculateStateSpaceMatrices() {
64  // Initialize matrices of state representation
65  mAStateSpace << -mLq / mTq0_t / mLq_t, 0, 0, -mLd / mTd0_t / mLd_t;
66  mBStateSpace << (mLq - mLq_t) / mTq0_t / mLq_t, 0.0, 0.0,
67  (mLd - mLd_t) / mTd0_t / mLd_t;
68  mCStateSpace << 0, 1 / mTd0_t;
69 
70  // Precalculate trapezoidal based matrices (avoids redundant matrix inversions in correction steps)
71  Math::calculateStateSpaceTrapezoidalMatrices(
72  mAStateSpace, mBStateSpace, mCStateSpace, mTimeStep, mAdTrapezoidal,
73  mBdTrapezoidal, mCdTrapezoidal);
74 }

76 void DP::Ph1::SynchronGenerator4OrderPCM::stepInPerUnit() {
77  // set number of iterations equal to zero
78  **mNumIter = 0;
79 
80  // store values currently at t=k for later use
81  mEdqtPrevStep = **mEdq_t;
82  mIdqPrevStep = **mIdq;
83  mVdqPrevStep = **mVdq;
84 
85  // update DQ-DP transforms according to mThetaMech
86  mDomainInterface.updateDQToDPTransform(**mThetaMech, mSimTime);
87  mDomainInterface.updateDPToDQTransform(**mThetaMech, mSimTime);
88 
89  // predict emf at t=k+1 (euler) using
90  (**mEdq_t) = Math::StateSpaceEuler(**mEdq_t, mAStateSpace, mBStateSpace,
91  mCStateSpace * **mEf, mTimeStep, **mVdq);
92 
93  // predict stator currents at t=k+1 (assuming Vdq(k+1)=Vdq(k))
94  (**mIdq)(0, 0) = ((**mEdq_t)(1, 0) - (**mVdq)(1, 0)) / mLd_t;
95  (**mIdq)(1, 0) = ((**mVdq)(0, 0) - (**mEdq_t)(0, 0)) / mLq_t;
96 
97  // convert currents to dp domain
98  (**mIntfCurrent)(0, 0) =
99  mDomainInterface.applyDQToDPTransform(**mIdq) * mBase_I_RMS;
100 }

108 void DP::Ph1::SynchronGenerator4OrderPCM::correctorStep() {
109 
110  // increase number of iterations
111  **mNumIter = **mNumIter + 1;
112 
113  // correct electrical vars
114  // calculate emf at j and k+1 (trapezoidal rule)
115  (**mEdq_t) = Math::applyStateSpaceTrapezoidalMatrices(
116  mAdTrapezoidal, mBdTrapezoidal, mCdTrapezoidal * **mEf, mEdqtPrevStep,
117  **mVdq, mVdqPrevStep);
118 
119  // calculate corrected stator currents at t=k+1 (assuming Vdq(k+1)=VdqPrevIter(k+1))
120  (**mIdq)(0, 0) = ((**mEdq_t)(1, 0) - (**mVdq)(1, 0)) / mLd_t;
121  (**mIdq)(1, 0) = ((**mVdq)(0, 0) - (**mEdq_t)(0, 0)) / mLq_t;
122 
123  // convert corrected currents to dp domain
124  (**mIntfCurrent)(0, 0) =
125  mDomainInterface.applyDQToDPTransform(**mIdq) * mBase_I_RMS;
126 
127  // stamp currents
128  mnaCompApplyRightSideVectorStamp(**mRightVector);
129 }

146 bool DP::Ph1::SynchronGenerator4OrderPCM::requiresIteration() {
147  if (**mNumIter >= mMaxIter) {
148  // maximum number of iterations reached
149  return false;
150  } else if (**mNumIter == 0) {
151  // no corrector step has been performed yet,
152  // convergence cannot be confirmed
153  return true;
154  } else {
155  // check voltage convergence according to tolerance
156  Matrix voltageDifference = **mVdq - mVdqPrevIter;
157  if (Math::abs(voltageDifference(0, 0)) > mTolerance ||
158  Math::abs(voltageDifference(1, 0)) > mTolerance)
159  return true;
160  else
161  return false;
162  }
163 }
