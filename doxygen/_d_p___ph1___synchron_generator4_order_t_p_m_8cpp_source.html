62 void DP::Ph1::SynchronGenerator4OrderTPM::calculateStateSpaceMatrices() {
63  mAStateSpace << -mLq / mTq0_t / mLq_t, 0, 0, -mLd / mTd0_t / mLd_t;
64  mBStateSpace << (mLq - mLq_t) / mTq0_t / mLq_t, 0.0, 0.0,

68  mAStateSpace, mBStateSpace, mCStateSpace, mTimeStep, mAdStateSpace,

94 void DP::Ph1::SynchronGenerator4OrderTPM::calculateConstantConductanceMatrix() {
95  Matrix resistanceMatrix = Matrix::Zero(2, 2);
96  resistanceMatrix(0, 0) = 0;
97  resistanceMatrix(0, 1) = (mA - mB) / 2.0;
98  resistanceMatrix(1, 0) = -(mA - mB) / 2.0;
99  resistanceMatrix(1, 1) = 0;
100 
101  SPDLOG_LOGGER_INFO(mSLog, "\nR_const [pu]: {}",
102  Logger::matrixToString(resistanceMatrix));
103 
104  resistanceMatrix = resistanceMatrix * mBase_Z;
105  SPDLOG_LOGGER_INFO(mSLog, "\nR_const [Ohm]: {}",
106  Logger::matrixToString(resistanceMatrix));
107 
108  mConductanceMatrixConst = resistanceMatrix.inverse();
109  SPDLOG_LOGGER_INFO(mSLog, "\nG_const [S]: {}",
110  Logger::matrixToString(mConductanceMatrixConst));

113 void DP::Ph1::SynchronGenerator4OrderTPM::mnaCompApplySystemMatrixStamp(
114  SparseMatrixRow &systemMatrix) {

116 
117  calculateConstantConductanceMatrix();
118

120  // Stamp conductance matrix

136  // set buttom right block

149 void DP::Ph1::SynchronGenerator4OrderTPM::stepInPerUnit() {
150  // set number of iteratios equal to zero
151  **mNumIter = 0;
152 
153  // update DP-DQ transforms
154  mDomainInterface.updateDQToDPTransform(**mThetaMech, mSimTime);
155  mDomainInterface.updateDPToDQTransform(**mThetaMech, mSimTime);
156 
157  // update varying resistance matrix part
158  Real DeltaTheta = **mThetaMech - mBase_OmMech * mSimTime;
159  mResistanceMatrixVarying(0, 0) = -(mA + mB) / 2.0 * sin(2 * DeltaTheta);
160  mResistanceMatrixVarying(0, 1) = (mA + mB) / 2.0 * cos(2 * DeltaTheta);
161  mResistanceMatrixVarying(1, 0) = (mA + mB) / 2.0 * cos(2 * DeltaTheta);
162  mResistanceMatrixVarying(1, 1) = (mA + mB) / 2.0 * sin(2 * DeltaTheta);
163  SPDLOG_LOGGER_DEBUG(mSLog, "\nR_var [pu] (t={:f}): {:s}", mSimTime,
164  Logger::matrixToString(mResistanceMatrixVarying));
165 
166  // predict stator current (linear extrapolation)
167  Matrix IdpPrediction = Matrix::Zero(2, 1);
168  IdpPrediction(0, 0) =
169  2 * (**mIntfCurrent)(0, 0).real() - mIdpTwoPrevStep(0, 0).real();
170  IdpPrediction(1, 0) =
171  2 * (**mIntfCurrent)(0, 0).imag() - mIdpTwoPrevStep(0, 0).imag();
172 
173  // calculate emf at t=k
174  (**mEdq_t)(0, 0) = (**mVdq)(0, 0) - (**mIdq)(1, 0) * mLq_t;
175  (**mEdq_t)(1, 0) = (**mVdq)(1, 0) + (**mIdq)(0, 0) * mLd_t;
176 
177  // calculate original history voltage of VBR model (trapezoidal rule)
178  mEh(0, 0) = mAd_t * (**mIdq)(1, 0) + mBd_t * (**mEdq_t)(0, 0);
179  mEh(1, 0) = mAq_t * (**mIdq)(0, 0) + mBq_t * (**mEdq_t)(1, 0) +
180  mDq_t * mEf_prev + mDq_t * (**mEf);
181 
182  // set to original history voltage in dp domain
183  **mEhMod = mDomainInterface.applyDQToDPTransform(mEh) * mBase_V_RMS;
184 
185  // add current prediction based component to modified history voltage of TPM model in dp domain
186  **mEhMod +=
187  -Complex(mBase_Z * (mResistanceMatrixVarying * IdpPrediction)(0, 0), 0);
188  **mEhMod +=
189  -Complex(0, mBase_Z * (mResistanceMatrixVarying * IdpPrediction)(1, 0));
190 
191  // store values currently at t=k for later use
192  mIdpTwoPrevStep = **mIntfCurrent;
193  mVdqPrevStep = **mVdq;
194  mEdqtPrevStep = **mEdq_t;
195 }

197 void DP::Ph1::SynchronGenerator4OrderTPM::mnaCompApplyRightSideVectorStamp(
198  Matrix &rightVector) {
199  if (mModelAsNortonSource) {
200  // Determine equivalent Norton current
201  mIhMod = Complex(mConductanceMatrixConst(0, 0) * (**mEhMod).real() +
202  mConductanceMatrixConst(0, 1) * (**mEhMod).imag(),
203  mConductanceMatrixConst(1, 0) * (**mEhMod).real() +
204  mConductanceMatrixConst(1, 1) * (**mEhMod).imag());
205  // Stamp Norton current
206  Math::setVectorElement(rightVector, matrixNodeIndex(0, 0), mIhMod);

208  // Stamp modified history voltage
209  Math::setVectorElement(rightVector, mVirtualNodes[1]->matrixNodeIndex(),
210  **mEhMod);

214 void DP::Ph1::SynchronGenerator4OrderTPM::correctorStep() {
215  // increase number of iterations
216  **mNumIter = **mNumIter + 1;
217 
218  // correct electrical vars
219  // calculate emf at j and k+1 (trapezoidal rule)
220  (**mEdq_t) = Math::applyStateSpaceTrapezoidalMatrices(
221  mAdStateSpace, mBdStateSpace, mCdStateSpace * **mEf, mEdqtPrevStep,
222  **mVdq, mVdqPrevStep);
223 
224  // calculate stator currents at j and k+1
225  (**mIdq)(0, 0) = ((**mEdq_t)(1, 0) - (**mVdq)(1, 0)) / mLd_t;
226  (**mIdq)(1, 0) = ((**mVdq)(0, 0) - (**mEdq_t)(0, 0)) / mLq_t;
227 
228  // convert corrected currents to dp domain
229  Complex IdpCorrectionComplex =
230  mDomainInterface.applyDQToDPTransform(**mIdq) * mBase_I_RMS;
231 
232  Matrix IdpCorrection = Matrix::Zero(2, 1);
233  IdpCorrection(0, 0) = IdpCorrectionComplex.real();
234  IdpCorrection(1, 0) = IdpCorrectionComplex.imag();
235 
236  // reset to original history voltage of VBR model in dp domain
237  **mEhMod = mDomainInterface.applyDQToDPTransform(mEh) * mBase_V_RMS;
238 
239  // add current correction based component to modified history voltage of TPM model in dp domain
240  **mEhMod +=
241  -Complex(mBase_Z * (mResistanceMatrixVarying * IdpCorrection)(0, 0), 0);
242  **mEhMod +=
243  -Complex(0, mBase_Z * (mResistanceMatrixVarying * IdpCorrection)(1, 0));
244 
245  // stamp currents
246  (**mRightVector).setZero();
247  mnaCompApplyRightSideVectorStamp(**mRightVector);
248 
249  // store value currently at j-1 for later use
250  mVdqPrevIter = **mVdq;
251 }

264 bool DP::Ph1::SynchronGenerator4OrderTPM::requiresIteration() {
265  if (**mNumIter >= mMaxIter) {
266  // maximum number of iterations reached
267  return false;
268  } else if (**mNumIter == 0) {
269  // no corrector step has been performed yet,
270  // convergence cannot be confirmed
271  return true;
272  } else {
273  // check voltage convergence according to tolerance
274  Matrix voltageDifference = **mVdq - mVdqPrevIter;
275  if (Math::abs(voltageDifference(0, 0)) > mTolerance ||
276  Math::abs(voltageDifference(1, 0)) > mTolerance)
277  return true;
278  else
279  return false;
280  }
281 }
