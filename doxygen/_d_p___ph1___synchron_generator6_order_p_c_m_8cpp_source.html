42 void DP::Ph1::SynchronGenerator6OrderPCM::specificInitialization() {
43 
44  // calculate state representation matrix
45  calculateStateSpaceMatrices();
46 
47  // initial voltage behind the transient reactance in the dq0 reference frame
48  (**mEdq_t)(0, 0) = (mLq - mLq_t) * (**mIdq)(1, 0);
49  (**mEdq_t)(1, 0) = -(mLd - mLd_t) * (**mIdq)(0, 0) + (**mEf);
50  (**mEdq_s)(0, 0) = (**mEdq_t)(0, 0) + (mLq_t - mLq_s) * (**mIdq)(1, 0);
51  (**mEdq_s)(1, 0) = (**mEdq_t)(1, 0) - (mLd_t - mLd_s) * (**mIdq)(0, 0);
52 
53  //
54  mEdqts << **mEdq_t, **mEdq_s;
55 
56  SPDLOG_LOGGER_INFO(mSLog,
57  "\n--- Model specific initialization ---"
58  "\nInitial Ed_t (per unit): {:f}"
59  "\nInitial Eq_t (per unit): {:f}"
60  "\nInitial Ed_s (per unit): {:f}"
61  "\nInitial Eq_s (per unit): {:f}"
62  "\nMax number of iterations: {:d}"
63  "\nTolerance: {:f}"
64  "\nSG Model: 6 Order (Anderson-Fouad) PCM"
65  "\n--- Model specific initialization finished ---",
66 
67  (**mEdq_t)(0, 0), (**mEdq_t)(1, 0), (**mEdq_s)(0, 0),
68  (**mEdq_s)(1, 0), mMaxIter, mTolerance);
69  mSLog->flush();
70 
71  mDomainInterface.setDPShiftFrequency(mBase_OmMech);
72 }

74 void DP::Ph1::SynchronGenerator6OrderPCM::calculateStateSpaceMatrices() {
75  // auxiliar constants
76  Real Bd_t = 1. / mTq0_t * (mLq - mLq_t);
77  Real Bq_t = 1. / mTd0_t * (mLd - mLd_t);
78  Real Bd_s = 1. / mTq0_s * (mLq_t - mLq_s);
79  Real Bq_s = 1. / mTd0_s * (mLd_t - mLd_s);
80 
81  // Initialize matrices of state representation of predictor step
82  mAStateSpace << -1. / mTq0_t, 0., 0., 0., 0., -1. / mTd0_t, 0., 0.,
83  1. / mTq0_s, 0., -1. / mTq0_s, 0., 0., 1. / mTd0_s, 0., -1. / mTd0_s;
84  mBStateSpace << 0., Bd_t, -Bq_t, 0., 0., Bd_s, -Bq_s, 0.;
85  mCStateSpace << 0., 1 / mTd0_t, 0., 0.;
86 
87  // Precalculate trapezoidal based matrices (avoids redundant matrix inversions in correction steps)
88  Math::calculateStateSpaceTrapezoidalMatrices(
89  mAStateSpace, mBStateSpace, mCStateSpace, mTimeStep, mAdTrapezoidal,
90  mBdTrapezoidal, mCdTrapezoidal);
91 }

93 void DP::Ph1::SynchronGenerator6OrderPCM::stepInPerUnit() {
94  // set number of iteratios equal to zero
95  **mNumIter = 0;
96 
97  // store values currently at t=k for later use
98  mIdqPrevStep = **mIdq;
99  mEdqtsPrevStep = mEdqts;
100 
101  // update DQ-DP transforms according to mThetaMech
102  mDomainInterface.updateDQToDPTransform(**mThetaMech, mSimTime);
103  mDomainInterface.updateDPToDQTransform(**mThetaMech, mSimTime);
104 
105  // predict emf at t=k+1 (euler) using
106  mEdqts = Math::StateSpaceEuler(mEdqts, mAStateSpace, mBStateSpace,
107  mCStateSpace * **mEf, mTimeStep, **mIdq);
108 
109  // predict armature currents for at t=k+1
110  (**mIdq)(0, 0) = (mEdqts(3, 0) - (**mVdq)(1, 0)) / mLd_s;
111  (**mIdq)(1, 0) = ((**mVdq)(0, 0) - mEdqts(2, 0)) / mLq_s;
112 
113  // convert currents to dp domain
114  (**mIntfCurrent)(0, 0) =
115  mDomainInterface.applyDQToDPTransform(**mIdq) * mBase_I_RMS;
116 }

124 void DP::Ph1::SynchronGenerator6OrderPCM::correctorStep() {
125  // corrector step (trapezoidal rule)
126  **mNumIter = **mNumIter + 1;
127 
128  // correct emf at t=k+1 (trapezoidal rule)
129  mEdqts = Math::applyStateSpaceTrapezoidalMatrices(
130  mAdTrapezoidal, mBdTrapezoidal, mCdTrapezoidal * **mEf, mEdqtsPrevStep,
131  **mIdq, mIdqPrevStep);
132 
133  // calculate corrected stator currents at t=k+1 (assuming Vdq(k+1)=VdqPrevIter(k+1))
134  (**mIdq)(0, 0) = (mEdqts(3, 0) - (**mVdq)(1, 0)) / mLd_s;
135  (**mIdq)(1, 0) = ((**mVdq)(0, 0) - mEdqts(2, 0)) / mLq_s;
136 
137  // convert corrected currents to dp domain
138  (**mIntfCurrent)(0, 0) =
139  mDomainInterface.applyDQToDPTransform(**mIdq) * mBase_I_RMS;
140 
141  // stamp currents
142  mnaCompApplyRightSideVectorStamp(**mRightVector);
143 }

159 bool DP::Ph1::SynchronGenerator6OrderPCM::requiresIteration() {
160  if (**mNumIter == 0)
161  // if no corrector step has been performed yet
162  return true;
163 
164  Matrix voltageDifference = **mVdq - mVdqPrevIter;
165  if (Math::abs(voltageDifference(0, 0)) > mTolerance ||
166  Math::abs(voltageDifference(1, 0)) > mTolerance) {
167  if (**mNumIter == mMaxIter) {
168  return false;
169  } else {
170  return true;
171  }
172  } else {
173  return false;
174  }
