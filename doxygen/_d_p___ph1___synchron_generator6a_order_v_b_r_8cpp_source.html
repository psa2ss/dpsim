33 void DP::Ph1::SynchronGenerator6aOrderVBR::specificInitialization() {
34 
35  // initial voltage behind the transient reactance in the dq reference frame
36  (**mEdq_t)(0, 0) = (mLq - mLq_t - mYq) * (**mIdq)(1, 0);
37  (**mEdq_t)(1, 0) =
38  (1 - mTaa / mTd0_t) * (**mEf) - (mLd - mLd_t - mYd) * (**mIdq)(0, 0);
39 
40  // initial dq behind the subtransient reactance in the dq reference frame
41  (**mEdq_s)(0, 0) = (**mVdq)(0, 0) - mLq_s * (**mIdq)(1, 0);
42  (**mEdq_s)(1, 0) = (**mVdq)(1, 0) + mLd_s * (**mIdq)(0, 0);
43 
44  // initialize history term behind the transient reactance
45  mEh_t(0, 0) = mAd_t * (**mIdq)(1, 0) + mBd_t * (**mEdq_t)(0, 0);
46  mEh_t(1, 0) = mAq_t * (**mIdq)(0, 0) + mBq_t * (**mEdq_t)(1, 0) +
47  mDq_t * (**mEf) + mDq_t * mEf_prev;
48 
49  SPDLOG_LOGGER_INFO(mSLog,
50  "\n--- Model specific initialization ---"
51  "\nSG model: 6th order type a (Marconato's model)"
52  "\nInitial Ed_t (per unit): {:f}"
53  "\nInitial Eq_t (per unit): {:f}"
54  "\nInitial Ed_s (per unit): {:f}"
55  "\nInitial Eq_s (per unit): {:f}"
56  "\n--- Model specific initialization finished ---",
57  (**mEdq_t)(0, 0), (**mEdq_t)(1, 0), (**mEdq_s)(0, 0),
58  (**mEdq_s)(1, 0));
59  mSLog->flush();
60 }

82  mEh_t(1, 0) = mAq_t * (**mIdq)(0, 0) + mBq_t * (**mEdq_t)(1, 0) +
83  mDq_t * (**mEf) + mDq_t * mEf_prev;
84

89  mCq_s * (**mEdq_s)(1, 0) + mDq_s * (**mEf) + mDq_s * mEf_prev;
