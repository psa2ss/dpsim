56 void EMT::Ph3::SynchronGenerator4OrderPCM::calculateStateSpaceMatrices() {
57  // Initialize matrices of state representation
58  mAStateSpace << -mLq / mTq0_t / mLq_t, 0.0, 0.0, 0, -mLd / mTd0_t / mLd_t,
59  0.0, 0, 0.0, 0.0;
60  mBStateSpace << (mLq - mLq_t) / mTq0_t / mLq_t, 0.0, 0.0, 0.0,
61  (mLd - mLd_t) / mTd0_t / mLd_t, 0.0, 0.0, 0.0, 0.0;
62  mCStateSpace << 0.0, 1. / mTd0_t, 0.0;
63  // Precalculate trapezoidal based matrices (avoids redundant matrix inversions in correction steps)
64  Math::calculateStateSpaceTrapezoidalMatrices(
65  mAStateSpace, mBStateSpace, mCStateSpace, mTimeStep, mAdTrapezoidal,
66  mBdTrapezoidal, mCdTrapezoidal);
67 }

69 void EMT::Ph3::SynchronGenerator4OrderPCM::stepInPerUnit() {
70  // set number of iterations equal to zero
71  **mNumIter = 0;
72 
73  // Predictor step (euler)
74 
75  // store values currently at t=k for later use
76  mEdq0tPrevStep = **mEdq0_t;
77  mIdq0PrevStep = **mIdq0;
78  mVdq0PrevStep = **mVdq0;
79 
80  // predict emf at t=k+1 (euler) using
81  (**mEdq0_t) = Math::StateSpaceEuler(**mEdq0_t, mAStateSpace, mBStateSpace,
82  mCStateSpace * **mEf, mTimeStep, **mVdq0);
83 
84  // predict stator currents at t=k+1 (assuming Vdq0(k+1)=Vdq0(k))
85  (**mIdq0)(0, 0) = ((**mEdq0_t)(1, 0) - (**mVdq0)(1, 0)) / mLd_t;
86  (**mIdq0)(1, 0) = ((**mVdq0)(0, 0) - (**mEdq0_t)(0, 0)) / mLq_t;
87  (**mIdq0)(2, 0) = 0.0;
88 
89  // convert currents into the abc domain
90  **mIntfCurrent = inverseParkTransform(**mThetaMech, **mIdq0);
91  **mIntfCurrent = **mIntfCurrent * mBase_I;
92 }

104 void EMT::Ph3::SynchronGenerator4OrderPCM::correctorStep() {
105  // increase number of iterations
106  **mNumIter = **mNumIter + 1;
107 
108  // correct electrical vars
109  // calculate emf at j and k+1 (trapezoidal rule)
110  (**mEdq0_t) = Math::applyStateSpaceTrapezoidalMatrices(
111  mAdTrapezoidal, mBdTrapezoidal, mCdTrapezoidal * **mEf, mEdq0tPrevStep,
112  **mVdq0, mVdq0PrevStep);
113 
114  // calculate corrected stator currents at t=k+1 (assuming Vdq(k+1)=VdqPrevIter(k+1))
115  (**mIdq0)(0, 0) = ((**mEdq0_t)(1, 0) - (**mVdq0)(1, 0)) / mLd_t;
116  (**mIdq0)(1, 0) = ((**mVdq0)(0, 0) - (**mEdq0_t)(0, 0)) / mLq_t;
117 
118  // convert currents into the abc domain
119  **mIntfCurrent = inverseParkTransform(**mThetaMech, **mIdq0);
120  **mIntfCurrent = **mIntfCurrent * mBase_I;
121 
122  // stamp currents
123  mnaCompApplyRightSideVectorStamp(**mRightVector);
124 }

143 bool EMT::Ph3::SynchronGenerator4OrderPCM::requiresIteration() {
144  if (**mNumIter >= mMaxIter) {
145  // maximum number of iterations reached
146  return false;
147  } else if (**mNumIter == 0) {
148  // no corrector step has been performed yet,
149  // convergence cannot be confirmed
150  return true;
151  } else {
152  // check voltage convergence according to tolerance
153  Matrix voltageDifference = **mVdq0 - mVdq0PrevIter;
154  if (Math::abs(voltageDifference(0, 0)) > mTolerance ||
155  Math::abs(voltageDifference(1, 0)) > mTolerance)
156  return true;
157  else
158  return false;
159  }
160 }

166 EMT::Ph3::SynchronGenerator4OrderPCM::parkTransform(Real theta,
167  const Matrix &abcVector) {
168  Matrix dq0Vector(3, 1);
169  Matrix abcToDq0(3, 3);
170 
171  // Park transform according to Kundur
172  abcToDq0 << 2. / 3. * cos(theta), 2. / 3. * cos(theta - 2. * PI / 3.),
173  2. / 3. * cos(theta + 2. * PI / 3.), -2. / 3. * sin(theta),
174  -2. / 3. * sin(theta - 2. * PI / 3.),
175  -2. / 3. * sin(theta + 2. * PI / 3.), 1. / 3., 1. / 3., 1. / 3.;
176 
177  dq0Vector = abcToDq0 * abcVector;
178 
179  return dq0Vector;
180 }

182 Matrix EMT::Ph3::SynchronGenerator4OrderPCM::inverseParkTransform(
183  Real theta, const Matrix &dq0Vector) {
184  Matrix abcVector(3, 1);
185  Matrix dq0ToAbc(3, 3);
186 
187  // Park transform according to Kundur
188  dq0ToAbc << cos(theta), -sin(theta), 1., cos(theta - 2. * PI / 3.),
189  -sin(theta - 2. * PI / 3.), 1., cos(theta + 2. * PI / 3.),
190  -sin(theta + 2. * PI / 3.), 1.;
191 
192  abcVector = dq0ToAbc * dq0Vector;
193 
194  return abcVector;
